index="am_cds" "Dispatching event to MQ for codeEvent"
| eval raw_clean = replace(_raw, "[\\r\\n]+", " ")
| rex field=raw_clean "payload\\s+(?<json_payload>\\{.*)"
| rex field=json_payload "\"cds:PLATFORM\"\\s*:\\s*\\{[^}]*\"value\"\\s*:\\s*\"(?<platform>[^\"]+)\""
| rex field=json_payload "\"cds:RequestStatus\"\\s*:\\s*\\{[^}]*\"value\"\\s*:\\s*\"(?<request_status>[^\"]+)\""
| where isnull(platform)
| table _time request_status platform json_payload
